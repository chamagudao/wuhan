    <script>
        let currentTel = '';
        // 模拟房源数据 后台审核通过才显示
        const houseList = [
            {id:1,title:"通山农村小院",addr:"咸宁通山乡下",price:"800元/月",desc:"独门独院 带菜地",tel:"13812345678",img:"https://picsum.photos/600/400?random=1"},
            {id:2,title:"武汉江夏农家院",addr:"江夏周边村庄",price:"1200元/月",desc:"交通便利 适合自住",tel:"13987654321",img:"https://picsum.photos/600/400?random=2"}
        ];

        // 手机号脱敏
        function hidePhone(phone){
            if(!phone) return '';
            return phone.slice(0,7) + '****';
        }

        // 渲染房源
        function renderList(){
            let html = '';
            houseList.forEach(item=>{
                html += `
                <div class="item">
                    <img src="${item.img}" class="item-img">
                    <div class="item-info">
                        <h3>${item.title}</h3>
                        <p>位置：${item.addr}</p>
                        <p>租金：${item.price}</p>
                        <p>简介：${item.desc}</p>
                        <p class="tel" onclick="openMask('${item.tel}')">联系电话：${hidePhone(item.tel)}</p>
                        <button class="btn-unlock" onclick="openMask('${item.tel}')">点击解锁完整号码</button>
                    </div>
                </div>
                `;
            });
            document.getElementById('list').innerHTML = html;
        }

        function openMask(tel){
            currentTel = tel;
            document.getElementById('mask').style.display = 'flex';
        }
        function closeMask(){
            document.getElementById('mask').style.display = 'none';
        }
        function unlockTel(){
            let name = document.getElementById('userName').value.trim();
            let wx = document.getElementById('userWechat').value.trim();
            if(!name || !wx){
                alert('请填写姓名和微信');
                return;
            }
            alert('解锁成功！完整电话：' + currentTel);
            closeMask();
        }

        renderList();
    </script>
